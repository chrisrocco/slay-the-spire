```typescript
export type ClientMessage =
  | { type: 'CREATE_ROOM'; nickname: string }
  | { type: 'JOIN_LOBBY'; roomCode: string; nickname: string }
  | { type: 'SELECT_CHARACTER'; character: string }
  | { type: 'TOGGLE_RULE'; rule: 'lastStand' | 'chooseYourRelic' }
  | { type: 'START_GAME' }
  | { type: 'PLAY_CARD'; cardId: string; targetIds?: string[] }
  | { type: 'END_TURN' }
  | { type: 'USE_POTION'; potionId: string; targetId?: string }
  | { type: 'SEND_CHAT'; text: string }
  | { type: 'RECONNECT'; roomCode: string; token: string };

export type ServerMessage =
  | { type: 'ROOM_CREATED'; roomCode: string; playerId: string; reconnectionToken: string }
  | { type: 'JOINED'; playerId: string; reconnectionToken: string }
  | { type: 'STATE_UPDATE'; state: GameState }
  | { type: 'LOBBY_UPDATE'; lobby: LobbyState }
  | { type: 'ERROR'; code: string; message: string }
  | { type: 'CHAT_MESSAGE'; playerId: string; text: string };
```

<task type="auto">
  <name>Task 2: Build WebSocket service and game store</name>
  <files>packages/client/src/services/websocket.ts, packages/client/src/stores/gameStore.ts</files>
  <action>
1. Create `packages/client/src/services/websocket.ts`:
   - Export `createGameConnection(url: string)` function that returns `{ connect, send, onMessage, connected }`.
   - `connect()` opens a WebSocket to the given URL.
   - `send(msg: ClientMessage)` serializes and sends if connected.
   - `onMessage(handler: (msg: ServerMessage) => void)` registers a message handler.
   - `connected` is a SolidJS signal (use `createSignal<boolean>(false)`).
   - Auto-reconnect on close with 2-second delay.
   - Parse incoming messages as JSON, cast to `ServerMessage`.
   - Import types from `@slay-online/shared`.

2. Create `packages/client/src/stores/gameStore.ts`:
   - Define `AppState` interface: `{ playerId, reconnectionToken, roomCode, game: GameState | null, lobby: LobbyState | null, chatMessages: Array<{playerId: string, nickname: string, text: string}>, error: string | null, phase: 'connecting' | 'lobby' | 'combat' }`.
   - Export `createAppStore()` that uses `createStore<AppState>(...)`.
   - Provide mutation functions: `setPlayerId`, `setReconnectionToken`, `setRoomCode`, `updateGame` (use `reconcile`), `updateLobby` (use `reconcile`), `addChat`, `setError`, `setPhase`.
   - Export `handleServerMessage(store, msg: ServerMessage)` that dispatches each message type:
     - `ROOM_CREATED` → store playerId, roomCode, reconnectionToken
     - `JOINED` → store playerId, reconnectionToken
     - `STATE_UPDATE` → updateGame, setPhase('combat')
     - `LOBBY_UPDATE` → updateLobby, setPhase('lobby')
     - `ERROR` → setError
     - `CHAT_MESSAGE` → addChat
   - Export a helper `getMyPlayer(state: AppState): PlayerState | undefined` that finds the current player by playerId.
  </action>
  <verify>`pnpm --filter @slay-online/client typecheck` passes. Both files import from @slay-online/shared without errors.</verify>
