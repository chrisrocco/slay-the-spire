```typescript
export const PlayerCardSchema = z.object({
  id: z.string(),
  name: z.string(),
  character: CharacterSchema.or(z.literal('colorless')),  // 'ironclad' | 'silent' | 'defect' | 'watcher' | 'colorless'
  rarity: RaritySchema,  // 'starter' | 'common' | 'uncommon' | 'rare' | 'colorless' | 'special'
  type: CardTypeSchema,  // 'Attack' | 'Skill' | 'Power' | 'Curse' | 'Status' | 'Daze'
  cost: z.number().int().min(0).max(6).or(z.literal('X')).or(z.literal('unplayable')),
  text: z.string(),
  upgraded: z.boolean().default(false),
  keywords: z.array(z.string()).default([]),
  exhaust: z.boolean().optional(),
  ethereal: z.boolean().optional(),
  retain: z.boolean().optional(),
  innate: z.boolean().optional(),
});
export type PlayerCard = z.infer<typeof PlayerCardSchema>;
```

**CRITICAL rules for extraction:**
- Use ONLY the .webp reference images as source of truth. Do NOT use video game card databases or wikis.
- Silent-specific board game mechanics to note carefully:
  - Poison in the board game is an HP loss (not damage) — it ignores block. Max combined 30 Poison tokens.
  - Shiv tokens: up to 5 max, each deals 1 damage when used. Shiv cards that create Shiv tokens vs. Shiv tokens that deal damage are different things — transcribe text exactly.
  - These differ from the video game — copy text exactly as printed on the card images.
- Base card IDs use snake_case of the card name (e.g., `neutralize`, `deadly_poison`).
- Upgraded card IDs: append `_upgraded` (e.g., `neutralize_upgraded`).
- `upgraded: false` for base cards, `upgraded: true` for upgraded cards.
- Set `exhaust: true` for cards with "Exhaust" text, `ethereal: true` for "Ethereal", etc.

**File structure:**
```typescript
import type { PlayerCard } from '../../schemas/cards.js';

export const silentCards = [
  {
    id: 'neutralize',
    name: 'Neutralize',
    character: 'silent' as const,
    rarity: 'starter' as const,
    type: 'Attack' as const,
    cost: 0,
    text: 'Deal 3 damage. Apply 1 Weak.',
    upgraded: false,
    keywords: ['weak'],
  },
  {
    id: 'neutralize_upgraded',
    name: 'Neutralize+',
    character: 'silent' as const,
    rarity: 'starter' as const,
    type: 'Attack' as const,
    cost: 0,
    text: 'Deal 4 damage. Apply 2 Weak.',
    upgraded: true,
    keywords: ['weak'],
  },
  // ... all remaining silent cards
] as const satisfies readonly PlayerCard[];

// O(1) lookup map — used by game engine in Phase 2
export const silentCardMap = Object.fromEntries(
  silentCards.map(c => [c.id, c])
) as Record<(typeof silentCards[number])['id'], PlayerCard>;
```
