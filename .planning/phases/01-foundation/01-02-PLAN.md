```typescript
export const PlayerCardSchema = z.object({
  id: z.string(),
  name: z.string(),
  character: CharacterSchema.or(z.literal('colorless')),  // 'ironclad' | 'silent' | 'defect' | 'watcher' | 'colorless'
  rarity: RaritySchema,  // 'starter' | 'common' | 'uncommon' | 'rare' | 'colorless' | 'special'
  type: CardTypeSchema,  // 'Attack' | 'Skill' | 'Power' | 'Curse' | 'Status' | 'Daze'
  cost: z.number().int().min(0).max(6).or(z.literal('X')).or(z.literal('unplayable')),
  text: z.string(),
  upgraded: z.boolean().default(false),
  keywords: z.array(z.string()).default([]),
  exhaust: z.boolean().optional(),
  ethereal: z.boolean().optional(),
  retain: z.boolean().optional(),
  innate: z.boolean().optional(),
});
export type PlayerCard = z.infer<typeof PlayerCardSchema>;
```

**CRITICAL rules for extraction:**
- Use ONLY the .webp reference images as source of truth. Do NOT use video game card databases or wikis — the board game has different damage values and mechanics.
- Extract both base and upgraded versions. Base card ID: e.g., `strike_r`. Upgraded card ID: `strike_r_upgraded`.
- `upgraded: false` for base cards, `upgraded: true` for upgraded cards.
- For `keywords`: include board-game-specific keywords present in card text (e.g., `'exhaust'`, `'ethereal'`, `'retain'`, `'innate'`, `'vulnerable'`, `'weak'`, `'strength'`, `'poison'`).
- For cards that say "Exhaust" at the bottom: set `exhaust: true`.
- For cards that say "Ethereal": set `ethereal: true`.
- For cards that say "Retain": set `retain: true`.
- For cards that say "Innate": set `innate: true`.
- For `cost: 'X'` cards: the cost field must be the literal string `'X'`, not a number.
- Enemy HP board note: This does NOT apply to player cards — ignore for this task.

**File structure:**
```typescript

export const ironcladCards = [
  {
    id: 'strike_r',

    rarity: 'starter' as const,
    type: 'Attack' as const,

    text: 'Deal 6 damage.',
    upgraded: false,
    keywords: [],
  },
  {
    id: 'strike_r_upgraded',
    name: 'Strike+',
    character: 'ironclad' as const,
    rarity: 'starter' as const,
    type: 'Attack' as const,
    cost: 1,
    text: 'Deal 9 damage.',
    upgraded: true,
    keywords: [],
  },
  // ... all remaining ironclad cards
] as const satisfies readonly PlayerCard[];

// O(1) lookup map — used by game engine in Phase 2
export const ironcladCardMap = Object.fromEntries(
  ironcladCards.map(c => [c.id, c])
) as Record<(typeof ironcladCards[number])['id'], PlayerCard>;
```
