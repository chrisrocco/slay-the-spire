export const PlayerCardSchema = z.object({
  id: z.string(),
  name: z.string(),
  character: CharacterSchema.or(z.literal('colorless')),
  rarity: RaritySchema,
  type: CardTypeSchema,
  cost: z.number().int().min(0).max(6).or(z.literal('X')).or(z.literal('unplayable')),
  text: z.string(),
  upgraded: z.boolean().default(false),
  keywords: z.array(z.string()).default([]),
  exhaust: z.boolean().optional(),
  ethereal: z.boolean().optional(),
  retain: z.boolean().optional(),
  innate: z.boolean().optional(),
});
export type PlayerCard = z.infer<typeof PlayerCardSchema>;
```

**CRITICAL rules for extraction:**
- Use ONLY the .webp reference images as source of truth. Do NOT use video game card databases or wikis.
- Defect-specific board game mechanics to transcribe accurately:
  - "Channel" means adding an Orb to the Orb slots. Board game Orbs do NOT rotate — any Orb can be directly evoked.
  - Lightning Orbs, Frost Orbs, and Dark Orbs each have different evoke and passive effects printed on the card text. Transcribe exactly.
  - "Evoke" triggers the evoke effect of a chosen Orb. This differs from video game (no rotation, no forced evoke order).
  - Any card that references "Devotion" is a video game mechanic NOT in the board game — flag it and copy whatever text is actually printed on the card image.
- Base card IDs: snake_case of name. Upgraded: append `_upgraded`.

    rarity: 'starter' as const,
    type: 'Skill' as const,

    text: 'Channel 1 Lightning.',
    upgraded: false,
    keywords: ['channel'],
  },
  {
    id: 'zap_upgraded',
    name: 'Zap+',
    character: 'defect' as const,
    rarity: 'starter' as const,
    type: 'Skill' as const,
    cost: 0,
    text: 'Channel 1 Lightning.',
    upgraded: true,
    keywords: ['channel'],
  },
  // ... all remaining defect cards
] as const satisfies readonly PlayerCard[];

export const defectCardMap = Object.fromEntries(
  defectCards.map(c => [c.id, c])
) as Record<(typeof defectCards[number])['id'], PlayerCard>;
```
