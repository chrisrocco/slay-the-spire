```typescript
// HP is either a fixed number or scales with player count (rulebook p.11)
export const EnemyHPSchema = z.union([
  z.number().int().positive(),
  z.object({ 1: z.number(), 2: z.number(), 3: z.number(), 4: z.number() }),
]);

// Three action pattern kinds
export const EnemyActionPatternSchema = z.discriminatedUnion('kind', [
  z.object({
    kind: z.literal('single'),
    description: z.string(),  // raw action text
  }),
  z.object({
    kind: z.literal('die'),
    description: z.string(),
    actions: z.record(z.string(), z.string()),  // die result "1"-"6" → action description
  }),
  z.object({
    kind: z.literal('cube'),
    description: z.string(),
    slots: z.array(z.object({
      text: z.string(),
      repeating: z.boolean(),  // false = gray slot (not repeated when cube loops)
    })),
  }),
]);

export const EnemyCardSchema = z.object({
  id: z.string(),
  name: z.string(),
  act: z.literal(1),
  category: z.enum(['encounter', 'elite', 'boss', 'summon', 'first_encounter']),
  hp: EnemyHPSchema,
  pattern: EnemyActionPatternSchema,
  specialAbilities: z.array(z.string()).default([]),  // Yellow text abilities
  summons: z.array(z.string()).default([]),            // IDs of enemies summoned at combat start
  rewards: z.object({
    gold: z.number().int().min(0).default(0),
    cardReward: z.boolean().default(false),
    potionReward: z.boolean().default(false),
    relicReward: z.boolean().default(false),
  }),
});
export type EnemyCard = z.infer<typeof EnemyCardSchema>;
```

**Key rules from the research:**
- The rulebook (not the video game) is the sole source of truth for enemy data.
- HP board enemies: Some enemies use the HP board with player-count scaling (rulebook page 11). Model these as `{ 1: N, 2: N, 3: N, 4: N }` objects.
- Single action enemies: `pattern: { kind: 'single', description: '...' }`
- Die action enemies: `pattern: { kind: 'die', description: '...', actions: { '1': '...', '2-3': '...', '4-6': '...' } }` — use string keys matching the die face groupings printed on the card.
- Cube action enemies: `pattern: { kind: 'cube', description: '...', slots: [{ text: '...', repeating: true }, { text: '...', repeating: false }] }` — gray slots have `repeating: false`.
- Special abilities (yellow text) go in `specialAbilities: ['...']` as plain strings.
- For enemies that summon other enemies at combat start, `summons: ['summoned_enemy_id']`.

Act 1 encounter categories to extract:
- Regular encounter enemies (drawn per player from encounter deck)
- First encounter enemy (if any — the mandatory first room)
- Summon enemies (summoned during combat, not placed from encounter deck directly)

**File structure:**

```typescript
import type { EnemyCard } from '../../schemas/enemies.js';

export const encounterEnemies = [
  {
    id: 'jaw_worm',
    name: 'Jaw Worm',
    act: 1 as const,
    category: 'encounter' as const,
    hp: 42,  // or { 1: 40, 2: 44, 3: 48, 4: 52 } for HP board enemies
    pattern: {
      kind: 'single' as const,
      description: 'Deal 11 damage. Gain 5 Block.',
    },
    specialAbilities: [],
    summons: [],
    rewards: {
      gold: 12,
      cardReward: false,
      potionReward: false,
      relicReward: false,
    },
  },
  // ... all Act 1 encounter enemies
] as const satisfies readonly EnemyCard[];

export const encounterEnemyMap = Object.fromEntries(
  encounterEnemies.map(e => [e.id, e])
) as Record<(typeof encounterEnemies[number])['id'], EnemyCard>;
```
